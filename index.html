Currying vs. Partial Application 

 Solution:

function curryPartial(fn, ...args) {
  if (args.length >= fn.length) {
    return fn(...args);
  }
  return function curried(...nextArgs) {
    return curryPartial(fn, ...args, ...nextArgs);
  };
